107// rename obsolete html
108fn rename_obsolete_html(dest_folder: &std::path::Path, vec_of_repo: &Vec<octocrab::models::Repository>) -> anyhow::Result<()> {
109    for entry in dest_folder.read_dir().log(pos!())?.flatten() {
110        if entry.file_name().to_string_lossy().ends_with(".html") {
111            let mut repo_exists = false;
112            for repo in vec_of_repo {
113                if format!("{}.html", &repo.name) == entry.file_name().to_string_lossy() {
114                    repo_exists = true;
115                    break;
116                }
117            }
118            if !repo_exists {
119                // rename the file
120                println!("Obsolete renamed: {}", &entry.file_name().to_string_lossy());
121                std::fs::rename(entry.path(), entry.path().with_extension("obsolete")).log(pos!())?;
122            }
123        }
124    }
125    Ok(())
126}

164fn get_article(body: &str) -> anyhow::Result<String> {
165    // for debug only -
166    std::fs::write("body_1.html", body).log(pos!())?;
167    let mut article: String;
168    if let Some(pos1) = crate::utils_mod::find_pos_end_data_before_delimiter(body, 0, "<article ") {
169        let pos2 = crate::utils_mod::find_pos_start_data_after_delimiter(body, 0, "</article>")
170            .ok_or_else(|| anyhow::anyhow!("delimiter is None"))
171            .log(pos!())?;
172        article = body[pos1..pos2].to_string();
173        article = remove_svg_octicon(&article).log(pos!())?.to_string();
174    } else {
175        // crazy story: the first README.md comes as a complete HTML with the article tag.
176        // but other README.md like https://github.com/bestia-dev-archived/cargo_crev_reviews_workspace/blob/main/cargo_crev_reviews/README.md
177        // comes with a stupid "richText" that will be expanded in javascript. Stupid react or some other framework.
178        let pos1 = crate::utils_mod::find_pos_end_data_before_delimiter(body, 0, r#""richText":""#)
179            .ok_or_else(|| anyhow::anyhow!("delimiter is None"))
180            .log(pos!())?;
181        let pos2 = crate::utils_mod::find_pos_start_data_after_delimiter(body, 0, r#"","renderedFileInfo"#)
182            .ok_or_else(|| anyhow::anyhow!("delimiter is None"))
183            .log(pos!())?;
184        article = body[pos1 + 12..pos2 - 19].to_string();
185        article = article
186            .replace(r#"\u003c"#, "<")
187            .replace(r#"\u003e"#, ">")
188            .replace(r#"\""#, r#"""#)
189            .replace(r#"\n"#, "\n")
190            .replace(r#"\\"#, r#"\"#);
191        // now it should look like the original HTML
192        article = remove_svg_octicon(&article).log(pos!())?.to_string();
193    }
194    // return article
195    img_src_modify(&article).log(pos!())
196}

198/// remove element svg class="octicon octicon-link">
199fn remove_svg_octicon(article: &str) -> anyhow::Result<String> {
200    use lol_html::{element, rewrite_str, RewriteStrSettings};
201    let element_content_handlers = vec![
202        // Rewrite insecure hyperlinks
203        element!("svg[class]", |el| {
204            let href = el.get_attribute("class").unwrap_or("".to_string());
205            if href.contains("octicon") {
206                el.remove();
207            }
208            Ok(())
209        }),
210    ];
211
212    let output = rewrite_str(
213        article,
214        RewriteStrSettings {
215            element_content_handlers,
216            ..RewriteStrSettings::default()
217        },
218    )
219    .log(pos!())?;
220
221    Ok(output)
222}

224/// if exists data-canonical-src then replace src
225fn img_src_modify(article: &str) -> anyhow::Result<String> {
226    use lol_html::{element, rewrite_str, RewriteStrSettings};
227    let element_content_handlers = vec![
228        // Rewrite insecure hyperlinks
229        element!("img[data-canonical-src]", |el| {
230            let canonical = el.get_attribute("data-canonical-src").unwrap_or("".to_string());
231            if !canonical.is_empty() {
232                el.set_attribute("src", &canonical).log(pos!())?;
233                el.remove_attribute("data-canonical-src");
234            }
235            Ok(())
236        }),
237    ];
238
239    let output = rewrite_str(
240        article,
241        RewriteStrSettings {
242            element_content_handlers,
243            ..RewriteStrSettings::default()
244        },
245    )
246    .log(pos!())?;
247
248    Ok(output)
249}

279async fn get_readme_body(repo: &octocrab::models::Repository) -> anyhow::Result<(String, String, String, String)> {
280    let repo_url = repo
281        .html_url
282        .as_ref()
283        .ok_or_else(|| anyhow::anyhow!("repo.html_url is None"))
284        .log(pos!())?;
285    println!("\n    Reading {}", repo_url);
286    // open the html
287    let body = reqwest::get(repo_url.clone()).await.log(pos!())?.text().await.log(pos!())?;
288
289    // get title and description
290    // They are already HTML encoded, because they come from a HTML
291    // find and parse: <title>GitHub - bestia-dev/github_readme_copy: Copy my public README.md files from Github in HTML format</title>
292    let long_title = get_long_title(&body).log(pos!())?;
293    let pos1 = long_title
294        .find(" - ")
295        .ok_or_else(|| anyhow::anyhow!("long_title.find is None"))
296        .log(pos!())?;
297    let pos2 = long_title
298        .find("/")
299        .ok_or_else(|| anyhow::anyhow!("long_title.find is None"))
300        .log(pos!())?;
301    let organization = long_title[pos1 + 3..pos2].to_string();
302    //println!("organization: {organization}");
303    let pos3 = long_title
304        .find(": ")
305        .ok_or_else(|| anyhow::anyhow!("long_title.find is None"))
306        .log(pos!())?;
307    let title = long_title[pos2 + 1..pos3].to_string();
308    //println!("title: {title}");
309    let description = long_title[pos3 + 2..].to_string();
310    //println!("description: {description}");
311    // check if the description of the project and the GitHub description is the same
312    let mut github_description = get_github_description(&body, &title).log(pos!())?.to_string();
313    // examples:  <meta name="description" content="Automation tasks coded in Rust language for the workflow of Rust projects - automation-tasks-rs/.github">
314    if let Some(pos1) = github_description.find(&format!(" - {organization}")) {
315        github_description = github_description[..pos1].to_string();
316    }
317    // example: <meta name="description" content="OBSOLETE: Library to encrypt/decrypt secrets. Contribute to automation-tasks-rs/cargo_auto_encrypt_secret_lib development by creating an account on GitHub.">
318    if let Some(pos1) = github_description.find(". Contribute to ") {
319        github_description = github_description[..pos1].to_string();
320    }
321    // example: <meta name="description" content="How to setup a development environment for rust in Win10 + WSL2 + VSCode for Longan nano GD32 Risc-V development board  - bestia-dev-archived/longan_nano_rust_wsl2_platformio_setup: How to setup a development environment for rust in Win10 + WSL2 + VSCode for Longan nano GD32 Risc-V development board">
322    println!("github_description: {github_description}");
323    if github_description != description {
324        println!();
325        println!("{RED}    Description different:{RESET}");
326        println!("{RED}    {title}{RESET}");
327        println!("{RED}    {description}{RESET}");
328        println!("{RED}    {github_description}{RESET}");
329        println!();
330    }
331
332    // find the magic link "Primary project README.md" it must be header2
333    // <h2 tabindex="-1" class="heading-element" dir="auto"><a href="https://github.com/bestia-dev/cargo_crev_reviews_workspace/tree/main/cargo_crev_reviews/README.md">Primary project README.md</a></h2>
334    // https://github.com/bestia-dev-archived/cargo_crev_reviews_workspace/blob/main/cargo_crev_reviews/README.md
335    let pos1 = body.find(r#"">Primary project README.md</a></h2>"#);
336    match pos1 {
337        None => Ok((body, title, description, organization)),
338        Some(pos1) => {
339            // extract the link
340            let delim2 = r#"<a href=""#;
341            let pos2 = body[..pos1]
342                .rfind(delim2)
343                .expect("The html {} has the phrase >Primary project README.md<, but before that there is no <a href=");
344            let pos3 = pos2 + delim2.len();
345            let link_url = &body[pos3..pos1];
346            println!("    Primary project: Reading {}", link_url);
347            // is there a redirect inside some javascript code??? Stupid.
348            // I  can download RAW, but then I have to transform into html.
349            let body = reqwest::get(link_url).await.log(pos!())?.text().await.log(pos!())?;
350            // for debug only - std::fs::write("body_1.html", &body).log(pos!())?;
351
352            Ok((body, title, description, organization))
353        }
354    }

394pub fn upload_github_readme(upload_url: &str) {
395    let source_folder = std::path::Path::new("tmp/github_readme");
396    let upload_url = format!("{upload_url}/github_readme");
397    // easy upload with rsync over SSH
398    println!(
399        "\nUploading from {}/ to server {}...\n",
400        source_folder.to_string_lossy(),
401        &upload_url
402    );
403    // the SSh key must be already ssh-add into the ssh-agent
404    // rsync -e ssh -avz --delete-after tmp/github_readme luciano_bestia@bestia.dev:/var/www/bestia.dev/github_readme
405    let mut rsync = std::process::Command::new("rsync");
406    rsync
407        .arg("-avz")
408        .arg("--delete-after")
409        .arg("--progress")
410        .arg("-e")
411        .arg("ssh") // tells rsync which port to use
412        // path must end with / to signal we want to copy the content and not the directory
413        .arg(format!("{}/", source_folder.to_string_lossy()))
414        // path must end with / to signal we want to copy the content and not the directory
415        .arg(&upload_url);
416
417    rsync.status().expect("rsync failed to execute");
418}
