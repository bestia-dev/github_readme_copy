3//! # decrypt github api token from file or use the oauth2 device workflow to get the access token and encrypt it and save into file
4//!
5//! Don't change this code, so it can be updated regularly with
6//! cargo auto update_automation_tasks_rs
7//! If you want to customize it, copy the code into main.rs and modify it there.
8//!
9//! ## Secrets
10//!
11//! In this module there will be a lot of work with secrets.  
12//! It is difficult to trust an external crate with your secrets.  
13//! External crates can get updated unexpectedly and change to malicious code.  
14//!
15//! ## Copy code instead of dependency crate
16//!
17//! It is best to have the Rust code under your fingertips when dealing with secrets.  
18//! Than you know, nobody will touch this code except of you.  
19//! You can copy this code directly into your codebase as a module,
20//! inspect and review it and know exactly what is going on.  
21//! The code is as linear and readable with comments as possible.
22//!
23//! ## Store encrypted secret to file
24//!
25//! The secrets will be encrypted with an ssh private key and stored in the `~/.ssh` folder.  
26//! This way the data is protected at rest in storage drive.  
27//!
28//! ## In memory protection
29//!
30//! This is a tough one! There is no 100% software protection of secrets in memory.  
31//! Theoretically an attacker could dump the memory in any moment and read the secrets.  
32//! There is always a moment when the secret is used in its plaintext form. This cannot be avoided.
33//! All we can do now is to be alert what data is secret and take better care of it.  
34//! Every variable that have secrets will have the word `secret` in it.
35//! When a variable is confusing I will use the word `plain` to express it is `not a secret`.
36//! To avoid leaking in logs I will use the `secrecy` crate. This is not 100% protection. It is important just to express intent when the secrets are really used.  
37//! `Secrecy` needs the trait `zeroize` to empty the memory after use for better memory hygiene.
38//! I will add the type names explicitly to emphasis the secrecy types used.
39//!
40//! ## encrypt_decrypt_with_ssh_key_mod
41//!
42//! This module depends on the generic module for encryption `encrypt_decrypt_with_ssh_key_mod.rs`. That module also needs to be copy and paste into your project.
43//!
44//! ## Other dependencies
45//!
46//! In `Cargo.toml` there are a group od dependencies needed for this to work. They are so generic that I don't expect any malware in them to be able to steal some usable secrets.  
47//!
48//! Beware that the versions of crates in `Cargo.toml` are not precisely pinpointed. In rust the symbol '=' means "the same major number equal or newer to". This means from one compilation to another, it can automatically change to a newer version without the programmer even noticing it.
49//!
50//! This is great if the newer version is solving some security issue. But this is super-bad if the newer version is malware supply chain attack. We have no idea how to distinguish one from another.
51//!
52//! Just to mention: there exists the trick to control the `Cargo.lock` file and forbid the change of the version number, but more times than not, you will not want to commit the lock file into the GitHub repository.
53//!
54//! ```toml
55//! [dependencies]
56//! anyhow="1.0.95"
57//! reqwest={version="0.12.12", features=["json","blocking"]}
58//! serde ={ version= "1.0.217", features=["std","derive"]}
59//! serde_json = "1.0.138"
60//! ssh-key = { version = "0.6.7", features = [ "rsa", "encryption","ed25519"] }
61//! ssh_agent_client_rs_git_bash = "0.0.19"
62//! rsa = { version = "0.9.7", features = ["sha2","pem"] }
63//! zeroize = {version="1.8.1", features=["derive"]}
64//! aes-gcm = "0.10.3"
65//! camino = "1.1.6"
66//! base64ct = {version = "1.6.0", features = ["alloc"] }
67//! secrecy = "0.10.3"
68//! chrono ="0.4.39"
69//! ```
70//!

117/// Start the github oauth2 device workflow
118/// It will use the private key from the .ssh folder.
119/// The encrypted file has the same file name with the ".enc" extension.
120/// Returns access_token to use as bearer for api calls
121pub fn get_github_secret_token() -> anyhow::Result<SecretString> {
122    let client_id = GITHUB_API_CONFIG
123        .get()
124        .ok_or_else(|| anyhow::anyhow!("GITHUB_API_CONFIG.get is None"))
125        .log(pos!())?
126        .client_id
127        .to_string();
128    let private_key_file_name = GITHUB_API_CONFIG
129        .get()
130        .ok_or_else(|| anyhow::anyhow!("GITHUB_API_CONFIG.get is None"))
131        .log(pos!())?
132        .github_api_private_key_file_name
133        .to_string();
134
135    println!("  {YELLOW}Check if the ssh private key exists.{RESET}");
136    let private_key_path_struct = ende::PathStructInSshFolder::new(private_key_file_name.clone())?;
137    if !private_key_path_struct.exists() {
138        eprintln!("{RED}Error: Private key {private_key_path_struct} does not exist.{RESET}");
139        println!("  {YELLOW}Create the private key in bash terminal:{RESET}");
140        println!(r#"{GREEN}ssh-keygen -t ed25519 -f {private_key_path_struct} -C "github api secret_token"{RESET}"#);
141        anyhow::bail!("Private key file not found.");
142    }
143
144    println!("  {YELLOW}Check if the encrypted file exists.{RESET}");
145    let encrypted_path_struct = ende::PathStructInSshFolder::new(format!("{private_key_file_name}.enc"))?;
146    if !encrypted_path_struct.exists() {
147        println!("  {YELLOW}Encrypted file {encrypted_path_struct} does not exist.{RESET}");
148        println!("  {YELLOW}Continue to authentication with the browser{RESET}");
149        let secret_access_token = authenticate_with_browser_and_save_file(&client_id, &private_key_path_struct, &encrypted_path_struct)?;
150        Ok(secret_access_token)
151    } else {
152        println!("  {YELLOW}Encrypted file {encrypted_path_struct} exist.{RESET}");
153        let plain_file_text = ende::open_file_b64_get_string(encrypted_path_struct.get_full_file_path())?;
154        // deserialize json into struct
155        let encrypted_text_with_metadata: ende::EncryptedTextWithMetadata = serde_json::from_str(&plain_file_text)?;
156
157        // check the expiration
158        let utc_now = chrono::Utc::now();
159        if encrypted_text_with_metadata.refresh_token_expiration.is_none() {
160            anyhow::bail!("refresh_token_expiration is None");
161        }
162        let refresh_token_expiration = chrono::DateTime::parse_from_rfc3339(
163            encrypted_text_with_metadata
164                .refresh_token_expiration
165                .as_ref()
166                .expect("The former line asserts this is never None"),
167        )?;
168        if refresh_token_expiration <= utc_now {
169            eprintln!("{RED}Refresh token has expired, start authentication_with_browser{RESET}");
170            let secret_access_token =
171                authenticate_with_browser_and_save_file(&client_id, &private_key_path_struct, &encrypted_path_struct)?;
172            return Ok(secret_access_token);
173        }
174        if encrypted_text_with_metadata.access_token_expiration.is_none() {
175            anyhow::bail!("access_token_expiration is None");
176        }
177        let access_token_expiration = chrono::DateTime::parse_from_rfc3339(
178            encrypted_text_with_metadata
179                .access_token_expiration
180                .as_ref()
181                .expect("The former line asserts this is never None"),
182        )?;
183        if access_token_expiration <= utc_now {
184            eprintln!("{RED}Access token has expired, use refresh token{RESET}");
185            let secret_response_refresh_token = decrypt_text_with_metadata(encrypted_text_with_metadata)?;
186            let secret_response_access_token: SecretBox<SecretResponseAccessToken> =
187                refresh_tokens(&client_id, secret_response_refresh_token.expose_secret().refresh_token.clone())?;
188            let secret_access_token = SecretString::from(secret_response_access_token.expose_secret().access_token.clone());
189            println!("  {YELLOW}Encrypt data and save file{RESET}");
190            encrypt_and_save_file(&private_key_path_struct, &encrypted_path_struct, secret_response_access_token)?;
191            return Ok(secret_access_token);
192        }
193        println!("  {YELLOW}Decrypt the file with the private key.{RESET}");
194        let secret_response_access_token = decrypt_text_with_metadata(encrypted_text_with_metadata)?;
195        let secret_access_token = SecretString::from(secret_response_access_token.expose_secret().access_token.clone());
196        Ok(secret_access_token)
197    }
198}

200fn authenticate_with_browser_and_save_file(
201    client_id: &str,
202    private_key_path_struct: &ende::PathStructInSshFolder,
203    encrypted_path_struct: &ende::PathStructInSshFolder,
204) -> anyhow::Result<SecretString> {
205    let secret_response_access_token: SecretBox<SecretResponseAccessToken> = authentication_with_browser(client_id)?;
206    let secret_access_token = SecretString::from(secret_response_access_token.expose_secret().access_token.clone());
207    println!("  {YELLOW}Encrypt data and save file{RESET}");
208
209    encrypt_and_save_file(private_key_path_struct, encrypted_path_struct, secret_response_access_token)?;
210    Ok(secret_access_token)
211}

213/// Oauth2 device workflow needs to be authenticated with a browser
214fn authentication_with_browser(client_id: &str) -> anyhow::Result<SecretBox<SecretResponseAccessToken>> {
215    // https://docs.github.com/en/apps/oauth-apps/building-oauth-apps/authorizing-oauth-apps#device-flow
216    // https://docs.github.com/en/apps/creating-github-apps/authenticating-with-a-github-app/generating-a-user-access-token-for-a-github-app#using-the-device-flow-to-generate-a-user-access-token
217    println!("  {YELLOW}Send request with client_id and retrieve device_code and user_code{RESET}");
218    println!("  {YELLOW}wait...{RESET}");
219
220    #[derive(serde::Serialize)]
221    struct RequestDeviceCode {
222        client_id: String,
223    }
224
225    #[derive(serde::Deserialize)]
226    struct ResponseDeviceCode {
227        device_code: String,
228        user_code: String,
229    }
230
231    let response_device_code: ResponseDeviceCode = reqwest::blocking::Client::new()
232        .post("https://github.com/login/device/code")
233        .header("Content-Type", "application/json")
234        .header("Accept", "application/json")
235        .json(&RequestDeviceCode {
236            client_id: client_id.to_owned(),
237        })
238        .send()?
239        .json()?;
240
241    println!("  {YELLOW}Copy this user_code:{RESET}");
242    println!("{GREEN}{}{RESET}", response_device_code.user_code);
243    println!("  {YELLOW}Open browser on and paste the user_code:{RESET}");
244    println!("{GREEN}https://github.com/login/device?skip_account_picker=true{RESET}");
245    println!("{BLUE}After the tokens are prepared on the server, press enter to continue...{RESET}");
246
247    let _user_input_just_enter_to_continue: String = inquire::Text::new("").prompt()?;
248
249    #[derive(serde::Serialize)]
250    struct RequestAccessToken {
251        client_id: String,
252        device_code: String,
253        grant_type: String,
254    }
255
256    println!("  {YELLOW}Send request with device_id and retrieve access tokens{RESET}");
257    println!("  {YELLOW}wait...{RESET}");
258    let secret_response_access_token: SecretBox<SecretResponseAccessToken> = SecretBox::new(
259        reqwest::blocking::Client::new()
260            .post(" https://github.com/login/oauth/access_token")
261            .header("Content-Type", "application/json")
262            .header("Accept", "application/json")
263            .json(&RequestAccessToken {
264                client_id: client_id.to_string(),
265                device_code: response_device_code.device_code.to_string(),
266                grant_type: "urn:ietf:params:oauth:grant-type:device_code".to_string(),
267            })
268            .send()?
269            .json()?,
270    );
271
272    Ok(secret_response_access_token)
273}

275/// use refresh token to get new access_token and refresh_token
276fn refresh_tokens(client_id: &str, refresh_token: String) -> anyhow::Result<SecretBox<SecretResponseAccessToken>> {
277    // https://docs.github.com/en/apps/creating-github-apps/authenticating-with-a-github-app/refreshing-user-access-tokens
278
279    #[derive(serde::Serialize)]
280    struct RequestWithRefreshToken {
281        client_id: String,
282        grant_type: String,
283        refresh_token: String,
284    }
285
286    println!("  {YELLOW}Send request with client_id and refresh_token and retrieve access tokens{RESET}");
287    println!("  {YELLOW}wait...{RESET}");
288    let secret_response_access_token: SecretBox<SecretResponseAccessToken> = SecretBox::new(Box::new(
289        reqwest::blocking::Client::new()
290            .post("https://github.com/login/oauth/access_token")
291            .header("Content-Type", "application/json")
292            .header("Accept", "application/json")
293            .json(&RequestWithRefreshToken {
294                client_id: client_id.to_owned(),
295                grant_type: "refresh_token".to_string(),
296                refresh_token,
297            })
298            .send()?
299            .json()?,
300    ));
301
302    Ok(secret_response_access_token)
303}

305/// encrypt and save file
306///
307/// The "seed" are just some random 32 bytes.
308/// The "seed" will be "signed" with the private key.
309/// Only the "owner" can unlock the private key and sign correctly.
310/// This signature will be used as the true passcode for symmetrical encryption.
311/// The "seed" and the private key path will be stored in plain text in the file
312/// together with the encrypted data in json format.
313/// To avoid plain text in the end encode in base64 just for obfuscate a little bit.
314fn encrypt_and_save_file(
315    private_key_path_struct: &ende::PathStructInSshFolder,
316    encrypted_path_struct: &ende::PathStructInSshFolder,
317    secret_response_access_token: SecretBox<SecretResponseAccessToken>,
318) -> anyhow::Result<()> {
319    let secret_string = SecretString::from(serde_json::to_string(&secret_response_access_token.expose_secret())?);
320
321    let (plain_seed_bytes_32bytes, plain_seed_string) = ende::random_seed_32bytes_and_string()?;
322
323    println!("  {YELLOW}Unlock private key to encrypt the secret symmetrically{RESET}");
324    let secret_passcode_32bytes: SecretBox<[u8; 32]> =
325        ende::sign_seed_with_ssh_agent_or_private_key_file(private_key_path_struct, plain_seed_bytes_32bytes)?;
326
327    println!("  {YELLOW}Encrypt the secret symmetrically {RESET}");
328    let encrypted_string = ende::encrypt_symmetric(secret_passcode_32bytes, secret_string)?;
329
330    // the file will contain json with 3 plain text fields: fingerprint, seed, encrypted, expiration
331
332    // calculate expiration minus 10 minutes or 600 seconds
333    let utc_now = chrono::Utc::now();
334    let access_token_expiration = utc_now
335        .checked_add_signed(chrono::Duration::seconds(
336            secret_response_access_token.expose_secret().expires_in - 600,
337        ))
338        .context("checked_add_signed")?
339        .to_rfc3339();
340    let refresh_token_expiration = utc_now
341        .checked_add_signed(chrono::Duration::seconds(
342            secret_response_access_token.expose_secret().refresh_token_expires_in - 600,
343        ))
344        .context("checked_add_signed")?
345        .to_rfc3339();
346
347    let encrypted_text_with_metadata = ende::EncryptedTextWithMetadata {
348        private_key_file_name: private_key_path_struct.get_file_name().to_string(),
349        plain_seed_string,
350        plain_encrypted_text: encrypted_string,
351        access_token_expiration: Some(access_token_expiration),
352        refresh_token_expiration: Some(refresh_token_expiration),
353        token_name: None,
354    };
355    let plain_file_text = serde_json::to_string_pretty(&encrypted_text_with_metadata)?;
356    // encode it just to obscure it a little bit
357    let file_text = ende::encode64_from_string_to_string(&plain_file_text);
358
359    let mut file = std::fs::File::create(encrypted_path_struct.get_full_file_path())?;
360    #[cfg(target_family = "unix")]
361    {
362        let metadata = file.metadata()?;
363        let mut permissions = metadata.permissions();
364        std::os::unix::fs::PermissionsExt::set_mode(&mut permissions, 0o600);
365    }
366    std::io::Write::write_all(&mut file, file_text.as_bytes())?;
367
368    println!("  {YELLOW}Encrypted text saved to file.{RESET}");
369
370    Ok(())
371}

373/// decrypt text with metadata
374///
375/// The encrypted file is encoded in base64 just to obfuscate it a little bit.  
376/// In json format in plain text there is the "seed", the private key path and the encrypted secret.  
377/// The "seed" will be "signed" with the private key.  
378/// Only the "owner" can unlock the private key and sign correctly.  
379/// This signature will be used as the true passcode for symmetrical decryption.  
380fn decrypt_text_with_metadata(
381    encrypted_text_with_metadata: ende::EncryptedTextWithMetadata,
382) -> anyhow::Result<SecretBox<SecretResponseAccessToken>> {
383    // the private key file is written inside the file
384    let private_key_path_struct = ende::PathStructInSshFolder::new(encrypted_text_with_metadata.private_key_file_name.clone())?;
385    if !camino::Utf8Path::new(private_key_path_struct.get_full_file_path()).exists() {
386        anyhow::bail!("{RED}Error: File {private_key_path_struct} does not exist! {RESET}");
387    }
388
389    let plain_seed_bytes_32bytes = ende::decode64_from_string_to_32bytes(&encrypted_text_with_metadata.plain_seed_string)?;
390    // first try to use the private key from ssh-agent, else use the private file with user interaction
391    let secret_passcode_32bytes: SecretBox<[u8; 32]> =
392        ende::sign_seed_with_ssh_agent_or_private_key_file(&private_key_path_struct, plain_seed_bytes_32bytes)?;
393    // decrypt the data
394    let decrypted_string = ende::decrypt_symmetric(secret_passcode_32bytes, encrypted_text_with_metadata.plain_encrypted_text)?;
395    // parse json to struct
396    let secret_response_access_token: SecretBox<SecretResponseAccessToken> =
397        SecretBox::new(Box::new(serde_json::from_str(decrypted_string.expose_secret())?));
398    Ok(secret_response_access_token)
399}

401pub(crate) fn send_to_github_api_with_secret_token(req: reqwest::blocking::RequestBuilder) -> anyhow::Result<serde_json::Value> {
402    // I must build the request to be able then to inspect it.
403    let req = req.bearer_auth(get_github_secret_token()?.expose_secret()).build()?;
404
405    // region: Assert the correct url and https
406    // It is important that the request coming from a external crate/library
407    // is only sent always and only to GitHub API and not some other malicious url,
408    // because the request contains the secret GitHub API secret_token.
409    // And it must always use https
410    let host_str = req.url().host_str().context("host_str")?;
411    assert!(
412        host_str == "api.github.com",
413        "{RED}Error: Url is not correct: {host_str}. It must be always api.github.com.{RESET}"
414    );
415    let scheme = req.url().scheme();
416    assert!(
417        scheme == "https",
418        "{RED}Error: Scheme is not correct: {scheme}. It must be always https.{RESET}"
419    );
420    // endregion: Assert the correct url and https
421
422    let reqwest_client = reqwest::blocking::Client::new();
423    let response_text = reqwest_client.execute(req)?.text()?;
424
425    let json_value: serde_json::Value = serde_json::from_str(&response_text)?;
426
427    // panic if "message": String("Bad credentials"),
428    if let Some(m) = json_value.get("message") {
429        if m == "Bad credentials" {
430            panic!("{RED}Error: Bad credentials for GitHub API. {RESET}");
431        }
432    }
433
434    // return
435    Ok(json_value)
436}
