5/// Initialize tracing to file logs/github_readme_copy.log.  \
6///
7/// The folder logs/ is in .gitignore and will not be committed.  
8pub fn tracing_init() -> anyhow::Result<()> {
9    let offset = time::UtcOffset::current_local_offset()?;
10    let timer = tracing_subscriber::fmt::time::OffsetTime::new(
11        offset,
12        time::macros::format_description!("[hour]:[minute]:[second].[subsecond digits:6]"),
13    );
14
15    // A filter consists of one or more comma-separated directives
16    // target[span{field=value}]=level
17    // Levels order: 1. ERROR, 2. WARN, 3. INFO, 4. DEBUG, 5. TRACE
18    // ERROR level is always logged.
19    // Add filters to GITHUB_README_COPY_LOG environment variable for a single execution:
20    // ```bash
21    // GITHUB_README_COPY_LOG="debug,hyper_util=info,reqwest=info" ./{package_name}
22    // ```
23    let filter = tracing_subscriber::EnvFilter::from_env("GITHUB_README_COPY_LOG");
24
25    let builder = tracing_subscriber::fmt()
26        .with_timer(timer)
27        .with_ansi(false)
28        .with_target(false)
29        .with_env_filter(filter);
30    if std::env::var("GITHUB_README_COPY_LOG").is_ok() {
31        // if GITHUB_README_COPY_LOG exists than enable tracing to file
32        let file_appender = tracing_appender::rolling::RollingFileAppender::builder()
33            .rotation(tracing_appender::rolling::Rotation::DAILY)
34            .filename_prefix("github_readme_copy")
35            .filename_suffix("log")
36            .build("logs")
37            .expect("initializing rolling file appender failed");
38        builder.with_writer(file_appender).init();
39    } else {
40        builder.init();
41    };
42
43    Ok(())
44}

46/// macro to get source code position to log errors before propagation
47///
48/// example:  read_to_string("x").log(pos!())?;
49#[macro_export]
50macro_rules! pos {
51    // `()` indicates that the macro takes no argument.
52    () => {
53        // The macro will expand into the contents of this block.
54        &format!("{}:{}:{}:", file!(), line!(), column!())
55    };
56}

58/// Trait to log the error from Result before propagation with ?.
59pub trait ResultLogError<T, E>: Sized {
60    fn log(self, file_line_column: &str) -> Self;
61}
62
63/// Implements LogError for anyhow::Result.
64impl<T, E: std::fmt::Debug> ResultLogError<T, E> for core::result::Result<T, E> {
65    fn log(self, file_line_column: &str) -> Self {
66        self.inspect_err(|err| tracing::error!("{} {:?}", file_line_column, err))
67    }
68}

72/// return the position of start of the delimited data after the delimiter
73pub fn find_pos_start_data_after_delimiter(md_text_content: &str, pos: usize, delimiter: &str) -> Option<usize> {
74    if let Some(pos_start_data) = find_from(md_text_content, pos, delimiter) {
75        let pos_start_data = pos_start_data + delimiter.len();
76        return Some(pos_start_data);
77    }
78    // return
79    None
80}

84    if let Some(pos_end_data) = find_from(md_text_content, pos, delimiter) {

93/// find from_pos
94pub fn find_from(text: &str, from_pos: usize, find: &str) -> Option<usize> {
95    let slice01 = text.get(from_pos..).expect("find_from out of index");
96    let option_location = slice01.find(find);
97    if let Some(location) = option_location {
98        // return Option with usize
99        Some(from_pos + location)
100    } else {
101        // return Option with none
102        option_location
103    }
